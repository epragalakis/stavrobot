Review what was discussed yesterday. First, run this query to get a compact summary of the day's conversations:

SELECT
  m.created_at::text AS time,
  m.role,
  CASE
    WHEN jsonb_typeof(m.content->'content') = 'array' THEN (SELECT string_agg(elem->>'text', ' ') FROM jsonb_array_elements(m.content->'content') AS elem WHERE elem->>'type' = 'text')
    ELSE m.content->>'content'
  END AS text,
  CASE
    WHEN m.role = 'assistant' AND jsonb_typeof(m.content->'content') = 'array' THEN (SELECT string_agg(elem->>'name' || '(' || LEFT((elem->'arguments')::text, 100) || ')', ', ') FROM jsonb_array_elements(m.content->'content') AS elem WHERE elem->>'type' = 'toolCall')
    ELSE NULL
  END AS tool_calls
FROM messages m
WHERE m.role IN ('user', 'assistant')
  AND m.created_at >= CURRENT_DATE - INTERVAL '1 day'
  AND m.created_at < CURRENT_DATE
ORDER BY m.created_at
